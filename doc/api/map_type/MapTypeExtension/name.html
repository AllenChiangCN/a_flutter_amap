String get name {
  switch (this) {
    case MapType.NORMAL:
      return 'NORMAL';
    case MapType.NIGHT:
      return 'NIGHT';
    case MapType.NAVI:
      return 'NAVI';
    case MapType.BUS:
      return 'BUS';
    case MapType.SATELLITE:
      return 'SATELLITE';
    default:
      return 'NORMAL';
  }
}
